        "Motivation": "Describe the attacker's motivation.",
        "Capabilities": [
          "List the attacker's capabilities or skills."
        ]
      },
      "Target": {
        "Type": "Specify the type of target (e.g., Web Application, Network Service).",
        "Vulnerabilities": [
          "List the vulnerabilities in the target system."
        ],
        "Security Measures": [
          "List security measures or configurations of the target."
        ]
      },
      "AttackSteps": [
        {
          "Step": 1,
          "Description": "Describe the first step of the attack.",
          "Tool used": "If an existed tool has been used add it"
          "Actions": [
            {
              "Hypotesis": "Describe if necessary the reason why the attacker is performing this action",
              "Action": "Describe the action performed",
              "Result": "Describe the result of the action",
            },
            // Add more actions if necessary
          ]
        },
        // Add more steps as needed
      ]
    }
}
